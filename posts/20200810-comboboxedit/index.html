public class ComboBoxItemEntity
    {
        private object _text = 0;
        private object _Value = "";

        /// <summary>
        /// 显示值
        /// </summary>
        public object Text
        {
            get { return this._text; }
            set { this._text = value; }
        }
        /// <summary>
        /// 对象值
        /// </summary>
        public object Value
        {
            get { return this._Value; }
            set { this._Value = value; }
        }

        public override string ToString()
        {
            return this.Text.ToString();
        }
    }
  1. 建立InitOptionSelectList初始化下拉绑定方法
private void InitOptionSelectList()
        {
            string[] strsOptions = { "归零", "定位" };

            try
            {
                this.comboBoxOption.Properties.BeginInit();
                this.comboBoxOption.Properties.Items.Clear();

                for (int i = 0; i < strsOptions.Length; i++)
                {
                    ComboBoxItemEntity comboBoxItemEntity = new ComboBoxItemEntity();
                    comboBoxItemEntity.Text = strsOptions[i].ToString();
                    comboBoxItemEntity.Value = i.ToString();
                    this.comboBoxOption.Properties.Items.Add(comboBoxItemEntity);
                    this.comboBoxOption.Properties.SelectedIndexChanged += new EventHandler(comboBoxOption_SelectedIndexChanged);
                }
            }
            catch(Exception ex)
            {
                XtraMessageBox.Show("初始化下拉错误!" + ex.Message.ToString(), "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
            finally
            {
                this.comboBoxOption.Properties.EndInit();
            }
        }

private void comboBoxOption_SelectedIndexChanged(object sender, EventArgs e)
        {
            ComboBoxItemEntity comboBoxItemEntity = new ComboBoxItemEntity();
            comboBoxItemEntity = (ComboBoxItemEntity)(sender as ComboBoxEdit).SelectedItem;
            this.intOption = int.Parse(comboBoxItemEntity.Value.ToString());
        }
